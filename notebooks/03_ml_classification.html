def plot_decision_boundary_2D(model, X, y, title):
    h = 0.02
    
    # Use ONLY first 2 features for visualization
    X_plot = X[:, :2]
    
    x_min, x_max = X_plot[:, 0].min() - 1, X_plot[:, 0].max() + 1
    y_min, y_max = X_plot[:, 1].min() - 1, X_plot[:, 1].max() + 1

    xx, yy = np.meshgrid(
        np.arange(x_min, x_max, h),
        np.arange(y_min, y_max, h)
    )

    # Add dummy third feature (CDR = 0 mean after scaling)
    dummy_cdr = np.zeros(xx.ravel().shape)

    grid = np.c_[xx.ravel(), yy.ravel(), dummy_cdr]

    Z = model.predict(grid)
    Z = Z.reshape(xx.shape)

    plt.figure()
    plt.contourf(xx, yy, Z, alpha=0.3)
    plt.scatter(X_plot[:, 0], X_plot[:, 1], c=y, edgecolors='k')
    plt.title(title)
    plt.xlabel("MMSE")
    plt.ylabel("nWBV")
    plt.show()
